# 自动文档系统

> 版本: v1.0 | 日期: 2026-04-15

---

## 1. 方案概述

### 1.1 目标

实现文档的 **"代码即文档"** 理念：

```
代码变更 ──▶ 自动生成 ──▶ 自动更新文档
                          │
                          ▼
                    1. API文档 (Swagger UI)
                    2. 组件文档 (Storybook)
                    3. 类型文档 (TSDoc)
                    4. 开发手册 (CI自动更新)
```

### 1.2 自动化层级

| 层级 | 自动化内容 | 工具 | 实时性 |
|------|-----------|------|--------|
| **L1 静态文档** | 架构图、部署方案 | 手动维护 | - |
| **L2 代码文档** | API接口、组件说明 | Swagger/Storybook | ✅ 实时 |
| **L3 运行时文档** | 实际API调用、数据流 | APM/链路追踪 | ✅ 实时 |
| **L4 AI生成文档** | 架构描述、流程说明 | AI辅助 | ⏳ 定期 |

## 2. 后端API文档自动化

### 2.1 技术方案

**SpringDoc OpenAPI** - 自动从代码注解生成API文档

### 2.2 Maven依赖

```xml
<!-- pom.xml 添加 -->
<dependency>
    <groupId>org.springdoc</groupId>
    <artifactId>springdoc-openapi-starter-webmvc-ui</artifactId>
    <version>2.3.0</version>
</dependency>
```

### 2.3 配置类

```java
package com.material.server.config;

import io.swagger.v3.oas.models.OpenAPI;
import io.swagger.v3.oas.models.info.Info;
import io.swagger.v3.oas.models.info.Contact;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class OpenApiConfig {

    @Bean
    public OpenAPI customOpenAPI() {
        return new OpenAPI()
            .info(new Info()
                .title("材料申请管理系统 API")
                .version("1.0.0")
                .description("提供材料申请、采购订单、导航菜单等核心功能的 RESTful API")
                .contact(new Contact()
                    .name("开发团队")
                    .email("support@example.com")));
    }
}
```

### 2.4 控制器注解示例

```java
package com.material.server.controller;

import com.material.server.entity.MaterialApplication;
import com.material.server.service.MaterialApplicationService;
import io.swagger.v3.oas.annotations.Operation;
import io.swagger.v3.oas.annotations.Parameter;
import io.swagger.v3.oas.annotations.tags.Tag;
import lombok.RequiredArgsConstructor;
import org.springframework.web.bind.annotation.*;

import java.util.HashMap;
import java.util.Map;

@RestController
@RequestMapping("/api/materials")
@RequiredArgsConstructor
@Tag(name = "材料申请", description = "材料申请单的增删改查")
public class MaterialApplicationController {

    private final MaterialApplicationService materialApplicationService;

    @GetMapping
    @Operation(summary = "查询材料申请列表", 
               description = "支持分页、筛选、排序查询")
    public Map<String, Object> list(
            @Parameter(description = "当前页码", example = "1")
            @RequestParam(defaultValue = "1") Integer current,
            
            @Parameter(description = "每页条数", example = "20")
            @RequestParam(defaultValue = "20") Integer size,
            
            @Parameter(description = "材料名称（模糊查询）")
            @RequestParam(required = false) String materialName,
            
            @Parameter(description = "申请人")
            @RequestParam(required = false) String applicant,
            
            @Parameter(description = "状态: draft/pending/approved/rejected")
            @RequestParam(required = false) String status,
            
            @Parameter(description = "部门")
            @RequestParam(required = false) String department) {

        // 业务逻辑...
        Map<String, Object> response = new HashMap<>();
        response.put("code", 200);
        response.put("message", "查询成功");
        return response;
    }

    @PostMapping
    @Operation(summary = "创建材料申请", description = "创建新的材料申请单")
    public Map<String, Object> create(
            @Parameter(description = "材料申请信息")
            @RequestBody MaterialApplication material) {
        
        materialApplicationService.save(material);
        
        Map<String, Object> response = new HashMap<>();
        response.put("code", 200);
        response.put("data", material.getId());
        response.put("message", "创建成功");
        return response;
    }

    @GetMapping("/{id}")
    @Operation(summary = "查询材料申请详情", description = "根据ID查询材料申请详情")
    public Map<String, Object> detail(
            @Parameter(description = "材料申请ID", required = true, example = "1")
            @PathVariable Long id) {
        
        MaterialApplication material = materialApplicationService.getById(id);
        
        Map<String, Object> response = new HashMap<>();
        response.put("code", 200);
        response.put("data", material);
        return response;
    }

    @PutMapping("/{id}")
    @Operation(summary = "更新材料申请", description = "更新指定ID的材料申请信息")
    public Map<String, Object> update(
            @Parameter(description = "材料申请ID", required = true)
            @PathVariable Long id,
            
            @Parameter(description = "材料申请信息")
            @RequestBody MaterialApplication material) {
        
        material.setId(id);
        materialApplicationService.updateById(material);
        
        Map<String, Object> response = new HashMap<>();
        response.put("code", 200);
        response.put("message", "更新成功");
        return response;
    }

    @DeleteMapping("/{id}")
    @Operation(summary = "删除材料申请", description = "删除指定ID的材料申请")
    public Map<String, Object> delete(
            @Parameter(description = "材料申请ID", required = true)
            @PathVariable Long id) {
        
        materialApplicationService.removeById(id);
        
        Map<String, Object> response = new HashMap<>();
        response.put("code", 200);
        response.put("message", "删除成功");
        return response;
    }
}
```

### 4.1 技术方案

**MyBatis-Plus Generator** - 从实体类自动生成数据库文档

### 4.2 Maven依赖

```xml
    <version>3.5.6</version>
</dependency>
<dependency>
    <groupId>org.freemarker</groupId>

</dependency>
```

### 4.3 代码生成器配置

```java
package com.material.server.generator;

import com.baomidou.mybatisplus.generator.FastAutoGenerator;
import com.baomidou.mybatisplus.generator.config.OutputFile;
import com.baomidou.mybatisplus.generator.config.rules.DateType;
import com.baomidou.mybatisplus.generator.engine.FreemarkerTemplateEngine;

import java.util.Collections;

public class DatabaseGenerator {

    public static void main(String[] args) {
        String projectPath = System.getProperty("user.dir");
        String outputDir = projectPath + "/target/generated-sources";
        
        FastAutoGenerator.create(
                "jdbc:mysql://localhost:3306/material_system?useUnicode=true&characterEncoding=utf-8",
                "root",
                "your_password"
            )
            .globalConfig(builder -> {
                builder.author("Auto Generator") // 设置作者
                    .outputDir(outputDir + "/java") // 输出目录
                    .dateType(DateType.TIME_PACK) // 使用 java.time 包
                    .commentDate("yyyy-MM-dd"); // 注释日期格式
            })
            .packageConfig(builder -> {
                builder.parent("com.material.server") // 设置父包名
                    .entity("entity") // 设置实体类包名
                    .mapper("mapper") // 设置 Mapper 包名
                    .service("service") // 设置 Service 包名
                    .serviceImpl("service.impl") // 设置 Service Impl 包名
                    .controller("controller") // 设置 Controller 包名
                    .pathInfo(Collections.singletonMap(
                        OutputFile.xml, 
                        outputDir + "/resources/mapper"
                    )); // 设置 XML 资源文件目录
            })
            .strategyConfig(builder -> {
                builder.addInclude("material_application", "nav_menu", 
                                   "favorite", "purchase_order", "user_preference") // 设置需要生成的表名
                    .addTablePrefix("") // 设置过滤表前缀
                    
                    // Entity 策略配置
                    .entityBuilder()
                    .enableLombok() // 启用 Lombok
                    .enableTableFieldAnnotation() // 启用字段注解
                    .logicDeleteColumnName("deleted") // 逻辑删除字段
                    .naming(NamingStrategy.underline_to_camel) // 数据库表名转实体类名
                    .columnNaming(NamingStrategy.underline_to_camel) // 数据库字段转实体类字段
                    
                    // Mapper 策略配置
                    .mapperBuilder()
                    .enableBaseResultMap() // 生成 ResultMap
                    .enableBaseColumnList() // 生成 SQL 片段
                    
                    // Service 策略配置
                    .serviceBuilder()
                    .formatServiceFileName("%sService") // Service 接口命名
                    .formatServiceImplFileName("%sServiceImpl") // Service 实现类命名
                    
                    // Controller 策略配置
                    .controllerBuilder()
                    .enableRestStyle() // 启用 REST 风格
                    .enableHyphenStyle(); // 启用连字符风格 URL
            })
            .templateEngine(new FreemarkerTemplateEngine()) // 使用 Freemarker 模板引擎
            .execute();
    }
}
```

## 4. SQL脚本

### 4.1 创建表

```sql
CREATE TABLE `material_application` (
  `id` bigint NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `application_no` varchar(50) NOT NULL COMMENT '申请单号',
  `material_name` varchar(200) NOT NULL COMMENT '材料名称',
  `spec` varchar(100) COMMENT '规格型号',
  `unit` varchar(20) COMMENT '单位',
  `quantity` decimal(15,4) DEFAULT 0 COMMENT '申请数量',
  `department` varchar(50) COMMENT '部门',
  `supplier` varchar(100) COMMENT '供应商',
  `description` text COMMENT '描述',
  `status` varchar(20) NOT NULL DEFAULT 'draft' COMMENT '状态: draft/pending/approved/rejected',
  `applicant` varchar(50) COMMENT '申请人',
  `apply_time` datetime COMMENT '申请时间',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP,
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `deleted` tinyint DEFAULT 0 COMMENT '逻辑删除',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_application_no` (`application_no`),
  KEY `idx_status_create_time` (`status`, `create_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='材料申请表';
```
